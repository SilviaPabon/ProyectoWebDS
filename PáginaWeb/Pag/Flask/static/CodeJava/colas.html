                public class ListNode {
                    public Object objeto;
                    public ListNode siguiente;
                    public ListNode anterior;
                    
                    public ListNode() {
                        this.objeto = null;
                        this.siguiente = null;
                        this.anterior = null;
                    }
                    public ListNode(Object objeto) {
                        this.objeto = objeto;
                        this.siguiente = null;
                        this.anterior = null;
                    }
                    
                    public ListNode(Object objeto, ListNode siguiente, ListNode anterior) {
                        this.objeto = objeto;
                        this.siguiente = siguiente;
                        this.anterior = anterior;
                    }

                    //Obtiene el valor del nombre de la columna designada en la fila actual de este objeto.
                    public Object getObject() {
                        return objeto;
                    }
                    
                    //Establece el valor del parámetro designado mediante el objeto y el tipo de destino dados.
                    public void setObject(Object objeto) {
                        this.objeto = objeto;
                    }
                    
                    //compara dos cadenas y devuelve verdadero si las cadenas son iguales y falso si no.
                    public boolean isEquals(Object objeto) {
                        if (this.getObject().toString().equals(objeto.toString())) {
                            return true;
                        }
                        return false;
                    }
                    
                    //compara dos cadenas (nodo) y devuelve verdadero si las cadenas son iguales y falso si no.
                    public boolean isEquals(ListNode node) {
                        if (this.toString().equals(node.toString())) {
                            return true;
                        }
                        return false;
                    }
                    
                    @Override
                    public String toString() {
                        return "ListNode{" +
                                "objeto=" + objeto +
                                ", anterior=" + anterior +
                                ", siguiente=" + siguiente +
                                '}';
                    }
                }

                public class ListQueue implements QueueInterface {
    
                    private int size;
                    public ListNode head;
                    public ListNode tail;
                  
                    public ListQueue(){
                        clear();
                    }
            

                    //Metodo que limpia la Cola en totalidad
                    @Override
                    public void clear() {
                        
                        head=null;
                        tail=null;
                        size=0;
                    }
                
                    //Metodo para verificar si la cabeza de la cola es 0
                    @Override
                    public boolean isEmpty() {
                        
                        return head == null;
                        
                    }
                
                    @Override
                    public Object extract() {
                        ListNode primero = head;
                        head = primero.siguiente;
                        head.anterior = null;
                        return primero.objeto;
                    }
                
                    //Metodo para insertar a la cola el elemento que ingrese el usuario.
                    @Override
                    public boolean insert(Object object) {
                        if (isEmpty()) {
                                head = new ListNode(object);
                                tail = head;
                            }else {
                                tail = new ListNode(object,null,tail);
                                tail.anterior.siguiente = tail;
                            }
                            size++;
                            return true;
                    }
                
                    //Metodo que retorna la talla de la cola
                    @Override
                    public int size() {
                        
                        return size;
                        
                    }

                    public boolean search(Object object) {
                        try {
                        ListNode ite = head;
                        while(ite!=null && ite.objeto !=object){
                            ite = ite.siguiente;

                        if (ite.objeto == object){
                            return true;
                        }
