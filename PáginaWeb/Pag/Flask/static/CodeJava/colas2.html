Clase ArrayQueue

            

                import java.util.Arrays;
                import java.util.Collections;
                
                public class ArrayQueue implements QueueInterface{
                    
                    private int size;
                    private Object[] array;
                
                    //Convierte la cola en un arreglo.
                    public ArrayQueue(int size) {
                        this.size = size;
                        this.array = new Object[(size > 0) ? size : 1];
                        clear();
                    }
                    
                
                    //Limpia la cola totalmente.
                    @Override
                    public void clear() {
                        for (int i = 0; i < array.length; i++) {
                            array[i] = null;
                        }
                    }
                
                    //Devuelve un valor de tipo booleano que indica si se ha inicializado una variable
                    @Override
                    public boolean isEmpty() {
                        return array[0] == null;
                    }
                
                    //Cambiar un objeto de posicion.
                    @Override
                    public Object extract() {
                        if (!isEmpty()) {
                            Object object = array[0];
                            array[0] = null;
                            return object;
                        } else {
                            return null;
                        }
                    }
                
                    //Metodo funcional para insertar un elemento a la cola.
                    @Override
                    public boolean insert(Object object) {
                        for (int i=0 ; i < array.length ; i++){
                            if (array[i] == null){
                                array[i] = object;
                                return true;
                            }
                        }
                        return false;
                    }
                
                    //Funciona para retornar el tamaño del arreglo actual.
                    @Override
                    public int size() {
                        return array.length;
                    }
                
                    //Buscar el objeto ingresado dentro del arreglo.
                    @Override
                    public boolean search(Object object) {
                        return Arrays.asList(array).contains(object);
                    }
                
                
                    //Invertir el orden de la cola.
                    @Override
                    public void reverse() {
                        Collections.reverse(Arrays.asList(array));
                    }
                    
                    //Ordenar e imprimir la cola o arreglo actual.
                    @Override
                    public void sort() {
                        Arrays.sort(array);
                    }
                
                    @Override
                    public String toString() {
                        String arreglo ="Array= ";
                        for (int i = 0; i< array.length ; i++){
                            arreglo+= array[i] + " ";
                        }
                        return arreglo;
                    }
                }
