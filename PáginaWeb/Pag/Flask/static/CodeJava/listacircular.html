                        public class ListNode {
                            private Object object;
                            public ListNode next;
                            
                            public ListNode(){
                                this.object=null;
                                this.next=null;
                            }
                            
                            public ListNode(Object object){
                                this.object=object;
                                this.next=null;
                            }
                            
                            public ListNode(Object object, ListNode next) {
                                this.object = object;
                                this.next = next;
                            }
                            
                            public Object getObject(){
                                return object;
                            }
                            
                            public void setObject (Object object){
                                this.object=object;
                            }
                            
                            public Object getNext(){
                                return next;
                            }
                            
                            public void setNext (ListNode next){
                                this.next = next;
                            }
                            
                            public boolean isEquals(Object object) {
                                if (this.getObject().toString().equals(object.toString())) {
                                    return true;
                                }
                                return false;
                            }
                        
                            public boolean isEquals(ListNode node) {
                                if (this.toString().equals(node.toString())) {
                                    return true;
                                }
                                return false;
                            }
                        
                            @Override
                            public String toString() {
                                return "ListNode{" +
                                        "object=" + object +
                                        ", next=" + next +
                                        '}';
                            }
                        }

                import java.util.Iterator;
               
                public class ListaCircular implements ListInterface, Iterable{
                    public ListNode head;
                    public ListNode tail;
                    private ListNode inode;
                    private int size;
                    
                    public ListaCircular(){
                        clear();
                    }

                    //Metodo para comprobar si la lista es vacia
                    @Override
                    public boolean isEmpty(){
                        return head == null;
                    }

                    //Es utilizado para retornar el tamaño de la lista, mas concretamente el total de valores, elementos
                    @Override
                    public int getSize(){
                        return size;
                    }

                    //Metodo que tiene como funcion limpiar la lista en su totalidad
                    @Override
                    public void clear() {
                        head = null;
                        tail = null;
                        size = 0;
                    }

                    //Retorna el valor mas encima de la lista, es decir la cabeza.
                    @Override
                    public Object getHead() {
                        return head;
                    }

                    //Retorna el valor mas abajo de la lista, es decir la cola.
                    @Override
                    public Object getTail() {
                        return tail;
                    }

                    //Metodo utilizador para agregar elementos a la lista por cabeza
                    @Override
                    public boolean insertHead(Object object){
                        try {
                            if (isEmpty()){
                                head = new ListNode(object);
                                tail = head;
                                tail.setNext(head);
                            }else{
                                head = new ListNode(object,head);
                                tail.setNext(head);
                            }
                            this.size++;
                            return true;
                        }catch (Exception e) {
                            return false;
                        }
                    }

                    //Metodo utilizador para agregar elementos a la lista por cola
                    @Override
                    public boolean insertTail (Object object){
                        try{
                            if (isEmpty()){
                                head = new ListNode(object);
                                tail = head;
                                tail.setNext(head);
                            }else{
                                tail.next = new ListNode(object);
                                tail = tail.next;
                                tail.setNext(head);
                            }
                            this.size++;
                            return true;
                        }catch (Exception e){
                            return false;
                        }
                    }

                    //Metodo para añadir elementos a la lista
                    @Override
                    public boolean add(Object object){
                        return insertTail(object);
                    }

                    //Metodo para buscar en la lista el elemento ingresado
                    @Override
                    public ListNode search(Object object) {
                        Iterator i = this.iterator();
                        ListNode inode;
                        while ((inode = i.next()) != null) {
                            if (inode.getObject().toString().equals(object.toString())) {
                                return inode;
                            }
                        }
                        return null;
                    }
                    @Override


                    public boolean insert(Object ob, Object object) {
                        try {
                            if (ob != null) {
                                ListNode node = this.search(ob);
                                if (node != null) {
                                    return insert(node, object);
                                } else {
                                    return false;
                                }
                            } else {
                                return false;
                            }
                        } catch (Exception e) {
                            return false;
                        }
                    }

                    //Metodo para agregar elemento en una posicion especifica
                    @Override
                    public boolean insert(ListNode node, Object object) {
                        try {
                            if (node.next==head) {
                                add(object);
                            } else {
                                ListNode newNode = new ListNode(object);
                                newNode.next = node.next;
                                node.next = newNode;
                            }
                            this.size++;
                            return true;
                        } catch (Exception e) {
                            return false;
                        }
                        
                    }

                    //Metodo implementado para eliminar un nodo de la lista.
                    @Override
                    public boolean remove(ListNode node) {
                        if (head != null){
                            if (head.next == head){
                                clear();
                            } 
                            else{
                            ListNode puntero = head;
                            while (puntero.next != node){
                                puntero = puntero.next;
                            }
                            ListNode temp = puntero.next;
                            puntero.next = temp.next;
                            temp.next = null;
                            }
                        }
                        size--;
                        return true;
                    }

                    //Metodo implementado para eliminar un elemento de la lista.
                    @Override
                    public boolean remove(Object object) {
                        if (head != null){
                            if (head.next == head){
                                clear();
                            } 
                            else{
                                ListNode node = search(object);
                                remove (node);
                            }
                        }
                        size--;
                        return false;
                    }
                   

                    //Metodo para retornar el siguiente nodo de la lista
                    @Override
                    public Object getNextTo(ListNode node){
                        return node.next;
                    }
                    
                    //Metodo para retornar el elemento antes de x nodo que esta en la lista
                    @Override
                    public ListNode getBeforeTo(ListNode node) {
                        ListNode beforenode = head;
                        while (beforenode != null && beforenode.next != node){
                            beforenode = beforenode.next;
                        }
                        return beforenode;
                    }
                    
                    //Hecho para verificar si el elemento tiene alguna subcadena o no.
                    @Override
                    public boolean contains(Object object) {
                        if (isEmpty()){
                            return false;
                        } else {
                            ListNode node = this.search(object);
                            if (node != null){
                                return true;
                            }else{
                                return false;
                            }
                        }
                    }

                    //Metodo para recorrer la lista y presentar sus elementos.
                    @Override
                    public Iterator iterator() {
                        inode = head;
                        return new Iterator<>() {
                            @Override
                            public boolean hasNext() {
                                return inode.next != head;
                            }
                            @Override
                            public ListNode next() {
                                if (inode != null) {
                                    ListNode tmp = inode;
                                    inode = inode.next;
                                    return tmp;
                                } else {
                                    return null;
                                }
                            }
                        };
                    }


                    //Metodo utilizado para devolver una vista de la lista entre los nodos ingresados
                    public ListaCircular sublist (ListNode from, ListNode to){
                        if (!isEmpty()){
                            ListaCircular sublist = new ListaCircular();
                            sublist.head = from;
                            sublist.tail = to;
                            ListNode ite = head;
                            while (ite.next != tail){
                                add(ite.next);
                            }
                        return sublist;
                        }else{
                            return null;
                        }
                    }
