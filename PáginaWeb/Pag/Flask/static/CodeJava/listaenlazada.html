                        public class ListNode {
            
                            private Object object;
                            public ListNode next;
                        
                            public ListNode() {
                                this.object = null;
                                this.next = null;
                            }
                        
                            public ListNode(Object object) {
                                this.object = object;
                                this.next = null;
                            }
                        
                            public ListNode(Object object, ListNode next) {
                                this.object = object;
                                this.next = next;
                            }
                        
                            public Object getObject() {
                                return object;
                            }
                        
                            public void setObject(Object object) {
                                this.object = object;
                            }
                        
                            public boolean isEquals(Object object) {
                                if (this.getObject().toString().equals(object.toString())) {
                                    return true;
                                }
                                return false;
                            }
                        
                            public boolean isEquals(ListNode node) {
                                if (this.toString().equals(node.toString())) {
                                    return true;
                                }
                                return false;
                            }
                        
                            @Override
                            public String toString() {
                                return "ListNode{" +
                                        "object=" + object +
                                        ", next=" + next +
                                        '}';
                            }
                        }

                import java.util.Iterator;
                import java.util.Arrays;
                import static java.lang.System.*;

                public class List implements ListInterface, Iterable {

                    private ListNode inode;
                    private int size;

                    public ListNode head;
                    public ListNode tail;

                    /**
                    * List
                    */
                    public List() {
                        clear();
                    }

                    
                    public List(Object object) {
                        add(object);
                    }

                    /* Metodo "es vacio" la lista se vuelve vacia por la cabeza, no contiene ningun elemento*/
                    @Override
                    public boolean isEmpty() {
                        return head == null;
                    }

                    
                    @Override
                    public int getSize() {
                        return size;
                    }

                    /* Metodo para limpiar la lista por completo, su tamñano es 0*/
                    @Override
                    public void clear() {
                        head = null;
                        tail = null;
                        size = 0;
                    }

                    /* Retorna el elemento que esta en la cabeza */
                    @Override
                    public Object getHead() {
                        return head;
                    }

                    /* Retorna el valor de la cola que esta en la lista */
                    @Override
                    public Object getTail() {
                        return tail;
                    }

                    /* Metodo buscar, para buscar dentro de la lista el elemento que desee */
                    @Override
                    public ListNode search(Object object) {
                        Iterator i = this.iterator();
                        ListNode inode;
                        while ((inode = i.next()) != null) {
                            if (inode.getObject().toString().equals(object.toString())) {
                                return inode;
                            }
                        }
                        return null;
                    }

                    /* Metodo para mostrar un dato, elemento, que esta en la cabeza */
                    @Override
                    public boolean add(Object object) {
                        return insertHead(object);
                    }

                    /* insertar elemento en cualquier posicion*/ 
                    @Override
                    public boolean insert(Object ob, Object object) {
                        try {
                            if (ob != null) {
                                ListNode node = this.search(ob);
                                if (node != null) {
                                    return insert(node, object);
                                } else {
                                    return false;
                                }
                            } else {
                                return false;
                            }
                        } catch (Exception e) {
                            return false;
                        }
                    }
                    
                    
                    @Override
                    public boolean insert(ListNode node, Object object) {
                        try {
                            if (node.next == null) {
                                insertTail(object);
                            } else {
                                ListNode newNode = new ListNode(object);
                                newNode.next = node.next;
                                node.next = newNode;
                            }
                            return true;
                        } catch (Exception e) {
                            return false;
                        }
                    }
                    
                    /* Insertar elemento por la cabeza */
                    @Override
                    public boolean insertHead(Object object) {
                        try {
                            if (isEmpty()) {
                                head = new ListNode(object);
                                tail = head;
                            } else {
                                head = new ListNode(object, head);
                            }
                            this.size++;
                            return true;
                        } catch (Exception e) {
                            return false;
                        }
                    }

                    /*
                    Insertar elemento por cola 
                    */
                    @Override
                    public boolean insertTail(Object object) {
                        try {
                            if (isEmpty()) {
                                head = new ListNode(object);
                                tail = head;
                            } 
                            else {
                                tail.next = new ListNode(object);
                                tail = tail.next;
                            }
                            this.size++;
                            return true;
                        } catch (Exception e) {
                            return false;
                        }
                    }

                    /*
                    Eliminar nodo
                    */
                    @Override
                    public boolean remove(ListNode node) {
                        if (head != null){
                            if (head.next == null){
                                clear();
                            } 
                            else{
                            ListNode puntero = head;
                            while (puntero.next != node){
                                puntero = puntero.next;
                            }
                            puntero.next = node.next;
                            node.next = null;
                            }
                        }
                        size--;
                        return true;
                    }

                    /*
                    Elimiar elemento ingresado
                    */
                    @Override
                    public boolean remove(Object object) {
                        if (head != null){
                            if (head.next == null){
                                clear();
                            } 
                            else{
                                ListNode node = search(object);
                                ListNode puntero = head;
                            while (puntero.next != node){
                                puntero = puntero.next;
                            }
                            ListNode temp = puntero.next;
                            puntero.next = temp.next;
                            temp.next = null;
                            }
                                
                        }
                        size--;
                        return false;
                    }

                    /* comprobar si el elemento tiene otra subcadena o no*/
                    @Override
                    public boolean contains(Object object) {
                        if (isEmpty()){
                            return false;
                        } else {
                            ListNode node = this.search(object);
                            if (node != null){
                                return true;
                            }else{
                                return false;
                            }
                        }
                    }

                    //se utiliza para devolver una matriz que contiene todos los elementos
                    @Override
                    public Object[] toArray(Object[] object) {
                        object = new Object[this.size];
                        Iterator i = this.iterator();
                        int cont = 0;
                        while(i.hasNext() && cont < this.size){
                            object[cont]=i.next();
                        }
                        return new Object[this.size];
                    }
                    
                    //se utiliza para devolver una matriz que contiene todos los elementos en el orden correcto
                    @Override
                    public Object[] toArray(){
                        Object[] object;
                        object = new Object[this.size];
                        Iterator i = this.iterator();
                        int cont = 0;
                        while(i.hasNext() && cont < this.size){
                            object[cont]=i.next();
                            cont++;
                        }
                        return object;
                    }
                    
                    //Metodo para retornar el elemento antes de x nodo que esta en la lista
                    @Override
                    public ListNode getBeforeTo(ListNode node) {
                        ListNode beforenode = head;
                        while (beforenode !=null && beforenode.next!=node){
                            beforenode = beforenode.next;
                        }
                        return beforenode;
                    }
                    @Override
                    public Object getBeforeTo(){
                        ListNode beforenode = this.head;
                        while (beforenode !=null && beforenode.next!=this.inode){
                            beforenode = beforenode.next;
                        }
                        return beforenode;
                    }

                    //Metodo para retornar el elemento despues de x nodo que esta en la lista
                    @Override
                    public Object getNextTo(ListNode node) {
                        Object ob = node.getObject();
                        ListNode inode = this.search(ob);
                        inode = inode.next;
                        return inode;
                    }
                    @Override
                    public Object getNextTo(){
                        ListNode temp = this.search(inode);
                        inode = temp.next;
                        return inode;
                    }

                    //Metodo utilizado para devolver una vista de la lista entre los nodos ingresados
                    @Override
                    public List subList(ListNode from, ListNode to) {
                        if (!isEmpty()){
                        List subList = new List();
                        subList.head = from;
                        subList.tail = to;
                        ListNode ite = head;
                            while (ite.next!=to && ite != ite.next){
                                subList.add(ite);
                                ite = ite.next;
                            }
                            return subList;
                        }else{
                            return null;
                        }
                    }

                    //Metodo utilizado para ordenar los elementos presentes en la lista en orden ascendente.
                    @Override
                    public List sortList() {
                        List sortList = new List();
                        Object[] arreglo = this.toArray();
                        Arrays.sort(arreglo);
                        int cont = 0;
                        while ( cont < arreglo.length){
                            sortList.add(arreglo[cont]);
                            cont++;
                        }
                        return sortList;
                    }

                    //Metodo para recorrer la lista y presentar sus elementos.
                    @Override
                    public Iterator iterator() {
                        inode = head;
                        return new Iterator<>() {
                            @Override
                            public boolean hasNext() {
                                return inode.next != null;
                            }
                            @Override
                            public ListNode next() {
                                if (inode != null) {
                                    ListNode tmp = inode;
                                    inode = inode.next;
                                    return tmp;
                                } else {
                                    return null;
                                }
                            }
                        };
                    }

                        if (node.next != null) {
                            rec(node.next);

                    }
