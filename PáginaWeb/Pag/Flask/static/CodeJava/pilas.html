Pila con Listas

Clase List Node

            

                public class ListNode {
    
                    Object object;
                    public ListNode next;
                
                    public ListNode() {
                        this.object = null;
                        this.next = null;
                    }
                
                    public ListNode(Object object) {
                        this.object = object;
                        this.next = null;
                    }
                
                    public ListNode(Object object, ListNode next) {
                        this.object = object;
                        this.next = next;
                    }
                
                    //Obtiene el valor del nombre de la columna designada en la fila actual de este objeto.
                    public Object getObject() {
                        return object;
                    }
                
                    //Establece el valor del parámetro designado mediante el objeto y el tipo de destino dados.
                    public void setObject(Object object) {
                        this.object = object;
                    }
                
                    //compara dos cadenas y devuelve verdadero si las cadenas son iguales y falso si no.
                    public boolean isEquals(Object object) {
                        if (this.getObject().toString().equals(object.toString())) {
                            return true;
                        }
                        return false;
                    }
                

                    //compara dos cadenas (nodo) y devuelve verdadero si las cadenas son iguales y falso si no.
                    public boolean isEquals(ListNode node) {
                        if (this.toString().equals(node.toString())) {
                            return true;
                        }
                        return false;
                    }
                
                    //Convertir el elemente en string
                    @Override
                    public String toString() {
                        return "ListNode{" +
                                "object=" + object +
                                ", next=" + next +
                                '}';
                    }
                }
