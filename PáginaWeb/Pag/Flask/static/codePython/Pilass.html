                from edu.List import List
                from edu.ListNode import ListNode
                
                class liststack(List):
                
                //Inicializador de la pila vacía
                    def __init__(self):
                        self.d = List()
                        self.top = None
                        self.size = 0
                
                        //Método para vaciar la Pila
                    def clear(self):
                        while self.isEmpty() == False:
                            tmp = self.top
                            self.top = self.top.next
                            tmp.next = None
                        self.size = 0
                
                        //Método para comprobar si la pila está vacia
                    def isEmpty(self):
                        return self.top == None
                
                        //Método para obtener el tamaño de la Pila
                    def getSize(self):
                        return self.size

                        //Método para mostrar el dato que se encuentra en la cima
                    def peek(self):
                        try:
                            if self.getSize() == 0:
                                return -1
                            else:
                                self.inode = self.top.object
                                return self.inode
                        except:
                            print("Exception")
                            return False
                
                            //Método para sacar un dato de la Pila
                    def pop(self):
                        try:
                            if self.getSize() == 0:
                                return -1
                            else:
                                self.inode = self.top.object
                                self.top = self.top.next
                                self.size -= 1
                                return self.inode
                        except:
                            print("Exception")
                            return False
                
                            //Método para agregar un dato a la Pila
                    def push(self, object):
                        try:
                            nodopila = ListNode(object)
                            if self.size == 0:
                                nodopila.next = None
                                self.top = nodopila
                            else:
                                nodopila.next = self.top
                                self.top = nodopila
                            print(object)
                            self.size -= 1
                            return True
                        except:
                            print("Exception")
                            return False
                
                            //Método para buscar un dato en la pila
                    def search(self, object):
                        super().search(object)
                        return True
                
                        //Método para buscar un dato en la Pila
                    def searchs(self, object):
                        try:
                            if self.size == 0:
                                return False
                            else:
                                inode = self.top.object
                                tmp = self.top.next
                                while inode != object and tmp != None:
                                    inode = tmp.next.object
                                return True
                        except:
                            print("Exception")
                            return False
